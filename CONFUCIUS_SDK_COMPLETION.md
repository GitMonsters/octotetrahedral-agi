## Usage Examples

### Basic Orchestration
```python
from ngvt_orchestrator import NGVTUnifiedOrchestrator, Task, OrchestratorConfig

config = OrchestratorConfig(max_iterations=10)
orchestrator = NGVTUnifiedOrchestrator(config)

task = Task(
    id="my_task",
    title="Multi-Model Inference",
    description="Process data through multiple models",
    max_iterations=10,
)

artifacts = await orchestrator.run_session(task)
print(f"Status: {artifacts['status']}")
print(f"Duration: {artifacts['elapsed_seconds']}s")
```

### Extension Development
```python
from ngvt_advanced_extensions import DataProcessingExtension, ExtensionMetadata

class MyExtension(DataProcessingExtension):
    def __init__(self):
        metadata = ExtensionMetadata(
            name="my_processor",
            version="1.0.0",
            author="you",
            description="Custom processor",
            category=ExtensionCategory.DATA_PROCESSING,
        )
        super().__init__(metadata)
    
    async def _process(self, data):
        return {"processed": len(data)}

# Register and execute
registry = AdvancedExtensionRegistry()
registry.register_extension(MyExtension())
result = await registry.execute_extension("my_processor", {"text": "hello"})
```

### Production Deployment
```python
from ngvt_production import ProductionDeploymentOrchestrator

orchestrator = ProductionDeploymentOrchestrator(
    config_file="production.yaml"
)

# Get system status
status = orchestrator.get_deployment_status()
print(f"Workers: {status['scaling']['current_workers']}")

# Health check
health = await orchestrator.perform_health_check()
print(f"Database: {health['database']['status']}")

# Create backup
backup = orchestrator.backup_manager.create_backup(
    "state", 
    get_current_state()
)
```
